###

- node / agent
  -     ,   PC
  - node Scripted Pipeline
- master
  - PC
- slave
  - (master) PC

###

```jsx
//
pipeline \{
  //   any
  agent any
  //   stages
  stages \{
      //  ('execute ls command')
      stage('execute ls command') \{
        //    steps
        steps \{
          // 'execute ls command'
          echo 'execute ls command'
          // shell  'ls -la'
          sh 'ls -la'
        \}
      \}
      //  ('Where do ls command execute?')
      stage('Where do ls command execute?') \{
        //    steps
        steps \{
          // 'Here is default.'
          echo 'Here is default.'
          // shell  'ls -la'
          sh 'ls -la'
          // 'you can define where to execute command.'
          echo 'you can define where to execute command.'
          // 'For example, I want to view root directory'
          echo 'For example, I want to view root directory'
          // shell  'ls -la /'
          sh 'ls -la /'
        \}
      \}
      //  ('Is there any other way?')
      stage('Is there any other way?') \{
         //    steps
        steps \{
           // 'Maybe sh is independent'
          echo 'Maybe sh is independent'
          // sh     cd / , pwd , ls -la
          sh 'cd /'
          sh 'pwd'
          sh 'ls -la'
          // 'If you want to execute multiple commands in one sh.'
          echo 'If you want to execute multiple commands in one sh.'
          // sh   cd / , pwd , ls -la
          sh '''cd /
          pwd
          ls -la'''
        \}
      \}
    \}
  \}
```
