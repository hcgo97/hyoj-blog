### @ManyToOne

Users

```sql
create table users (
(
    id           bigint auto_increment primary key,
		user_id      varchar(20)  null,
		user_name    varchar(20)  null,
    company_seq  varchar(10)  null, // FK
    created_at   datetime     null,
    updated_at   datetime     null,
    is_deleted   varchar(255) null,

		constraint users_company_company_seq_fk
        foreign key (company_seq) references company (company_seq)
);
```

```java
public class Users \{

		// Users -\> Many
		// Company -\> One

		@ManyToOne(fetch = FetchType.LAZY)
		@JoinColumn(name = "company_seq")
		private Company company;
\}
```

### @OneToOne

```sql
create table app (
(
    id           bigint auto_increment primary key,
		app_key      varchar(40)  null,
		app_name     varchar(20)  null,
    company_seq  varchar(10)  null, // FK
    created_at   datetime     null,
    updated_at   datetime     null,
    is_deleted   varchar(255) null,

		constraint app_company_company_seq_fk
        foreign key (company_seq) references company (company_seq)
);
```

```java
public class App \{

		@OneToOne
		@JoinColumn(name = "company_seq")
		private Company company;
\}

public class Company \{

		// @OneToMany       .

		@OneToOne(fetch = FetchType.LAZY, mappedBy = "company")
		private App app;
\}
```
