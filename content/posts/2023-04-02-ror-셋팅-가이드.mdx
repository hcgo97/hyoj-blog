# Dockerfile

FROM ruby:3.0.3-alpine3.15 AS bundles

RUN apk add --no-cache --update build-base mariadb-dev

WORKDIR /app
ADD Gemfile* /app/

ENV BUNDLER_VERSION='2.2.32'
RUN gem install bundler --no-document -v $BUNDLER_VERSION && \\
    bundle install

FROM ruby:3.0.3-alpine3.15 AS app

RUN apk add --no-cache --update tzdata mariadb-dev less

WORKDIR /app
EXPOSE 3000
ENV TZ Asia/Seoul
ENV MALLOC_ARENA_MAX=2

COPY --from=bundles /root/.bundle /root/.bundle
COPY --from=bundles /usr/local/bundle /usr/local/bundle

ADD . /app/

# frozen_string_literal: true

class GoogleJsonStyleGuideAdapter \< ActiveModelSerializers::Adapter::Base
  def serializable_hash(options = \{ type: :api \})
    options = serialization_options(options)
    options[:fields] ||= instance_options[:fields]
    @serialized = serializer.serializable_hash(instance_options, options, self)

    api_version = options.fetch(:api_version, ActiveModelSerializers.config.default_api_version)
    params = options.fetch(:params, \{\})

    @document = \{
      api_version: api_version,
      params: params
    \}

    if options[:type] == :error
      @error = @serialized
      build_error
    else
      build_data
    end

    self.class.transform_key_casing!(@document, instance_options)
  end

  protected
    def build_error
      # message      HTTP Status
      # code:
      raise "in Error hash, :status is mandatory." unless @error[:status]
      @error[:status] = Rack::Utils.status_code(@error[:status])
      @error[:code] = @error[:code]
      @error[:message] ||= Rack::Utils::HTTP_STATUS_CODES[@error[:status]]
      @error[:debug_context] = @error[:context]
      @document[:error] = @error.slice(:status, :code, :message, :context)
    end

    def build_data
      if @serialized.is_a? Array
        @document[:data] = datum
      else
        @document[:data] = @serialized
      end
    end

    def datum
      # object  kaminari
      #    serializer.object
      total_count = if serializer.object.respond_to?(:total_count)
        serializer.object.total_count
      else
        @serialized.size
      end

      @document[:data] = \{
        total_items: total_count,
        size: @serialized.size,
        items: @serialized
      \}
    end
end

    # frozen_string_literal: true

    module Authenticatable
      extend ActiveSupport::Concern

      attr_reader :current_user

      def auth_payload
        @payload
      end

      def authenticate_user
        # decode =\> camelcase
        @current_user ||= if access_token
          algorithms = [Rails.configuration.x.auth.signing_algorithm]
          key = Rails.configuration.x.auth.secret_key
          decoded = JWT.decode(access_token, key, true, \{ algorithms: algorithms \})
          @payload = payload = CaseTransform.underscore(decoded.first || \{\})

          if payload.fetch("type", nil) == "ACCESS_TOKEN"
            user_uid = payload.fetch("user_id", nil)
            @user = User.find_by(uid: user_uid)
            status
          else
            nil
          end
        else
          nil
        end
      end

      def status
        case @user.status.to_sym
        when :complete, :block
          @user
        else
          nil
        end
      end

      def authenticate_admin
        if access_token
          key = Rails.configuration.x.oauth.hmac
          aud = Rails.configuration.x.oauth.client_id
          JWT.decode(access_token, key, true, verify_iss: true, iss: ENV.fetch("ARM_JWT_ISSUER"),
                    verify_aud: true, aud: aud)
        else
          raise StandardError.new
        end
      end

      def authenticate_user!
        authenticate_user
        if current_user.nil?
          Rails.logger.fatal "Invalid Access Token"
          render error: \{ status: :unauthorized, code: I18n.t("base_errors.invalid_access_token.code"),
                          message: I18n.t("base_errors.invalid_access_token.message"), context: "Invalid Access Token" \}
        else
          if current_user.status == "block"
            Rails.logger.fatal "User Is Blocked"
            raise UserBlockedError.new
          end
        end
      rescue StandardError
        Rails.logger.info "Access Token Parse Error"
        render error: \{ status: :unauthorized, code: I18n.t("base_errors.not_parse_access_token.code"),
                        message: I18n.t("base_errors.not_parse_access_token.message"), context: "Access Token Parse Error" \}
      end

      def authenticate_admin!
        authenticate_admin
      rescue StandardError

    # frozen_string_literal: true

    module ParameterAcceptable
      extend ActiveSupport::Concern

      included do
        def self.parameter_to_send
          @@parameter_to_send
        end

        def self.define_params(parameter_to_send, definition)
          @@parameter_to_send = parameter_to_send
          @@defined_params = definition
        end

        def self.defined_params
          @@defined_params
        end

        def self.ref_params
          definition = yield
          Proc.new \{ param_filters(definition) \}
        end
      end

      attr_accessor :permitted_params

      def accept_params(*filter_names)
        params = send(self.class.parameter_to_send)
        @permitted_params = params.permit(param_filters(filter_names))
        @permitted_params
      end

      def param_filters(filter_names)
        filter_names.collect do |filter_name|
          defined_params = self.class.defined_params

          filter = defined_params.fetch(filter_name, [filter_name])
          filter = instance_eval(&filter) if filter.respond_to?(:call)

          filter
        end
      end
    end

### Serializer

모델이 최종적으로 JSON포맷으로 변경해 Response로 보내기 위해 Serializer를 활용합니다.

관련Gem:
active_record_serializer ([https://github.com/rails-api/active_model_serializers](https://github.com/rails-api/active_model_serializers))

가이드문서: [https://github.com/rails-api/active_model_serializers/tree/v0.10.6/docs](https://github.com/rails-api/active_model_serializers/tree/v0.10.6/docs)

컨트롤러에서 Serializer를 지정하지 않으면 자동지정됩니다.

예를들어 모델이 `User` 라면 `UserSerializer` 가 로드됩니다.

ApplicationSerializer 가 테이블의 `uid` 를 JSON에서는 `id` 로 변경됩니다. (`uid` 가 있는경우만)

```

# uid가 있는 경우 자동으로 :id 로 변경함

attribute :uid, key: :id, if: -\> () \{ object.respond_to?(:uid) \}

```

attribute 또는 attributes 는 항상 노출, has_* 은 명시적으로 노출을 지시해야 노출합니다.

이를 이용해 has_one 관계를 설정하면 컨트롤러 액션마다 서로 다른 Response를 줄 수 있습니다.

이 기능를 통해 서로 다른 Response를 주는 경우(예: 관리자 전용 API와 앱전용 API의 Response분리)나 불필요한 데이터베이스 쿼리를 처리하지 않을 수 있습니다.

```

class ClientSerializer \< ApplicationSerializer
attributes :name, :check_in_count_of_company, :label, :notes # 상시 노출할 항목
has_one :place, serializer: PlaceSerializer # 액션의 필요에 따라 노출할 항목
end

```

컨트롤러에서 노출항목 결정방법

```

render api: client, includes: %w(place, last_meeting_organizer, last_meeting_organizer.team) # last_meeting_organizer.team 은 organizer(User의 팀까지 include, nested 방식)

```
